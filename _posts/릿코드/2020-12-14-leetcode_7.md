class Solution:
    def __init__(self):
        self.ret = 0
        
    def sol(self,idx,ret,slist):
        #print(slist,idx,ret)
        lenlist= len(slist)
        if lenlist == 1:
            return slist[0] + ret, 0
        if idx == 0:
            ret += slist[0] * slist[1]
        elif idx == lenlist - 1:
            ret += slist[idx] * slist[idx - 1]
        else:
            ret += slist[idx] * slist[idx - 1] * slist[idx + 1]
        
        slist.pop(idx)
        comp = 0
        comp_idx= 0
        for i in range(len(slist)):
            comp1, cidx = self.sol(i,ret,deepcopy(slist))
            #print(123)
            if comp1 > comp:
                comp = comp1
                comp_idx = cidx
        return comp, comp_idx
            
    def maxCoins(self, nums: List[int]) -> int:
        ret = 0
        comp = 0
        self.nums = nums
        #print(nums)
        slist = nums
        for i in range(len(nums)):
            comp1,idx = self.sol(i,ret,deepcopy(nums))
            if comp1 > comp:
                comp = comp1
        return comp
